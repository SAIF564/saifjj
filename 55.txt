      channel: room,
      onoff: "On"
    };

      if (err) console.error(err);

        onoff: "Off"
      };

      return [
        message.channel.send(`**The log Is __????__ !**`),

      return [
        message.channel.send(`**The log Is __??????__ !**`),

      ];

      if (err)
        console.error(err).catch(err => {
          console.error(err);
        });
    });
